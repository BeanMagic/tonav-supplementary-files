249 template <typename RawEventT>
250 void Tracker<RawEventT>::run()
251 {
252  // This method is rather big. All filters and variables are local
253  // because their type is deducted by "auto". It would be difficult
254  // to create an std::tuple<LotOfFilters> as a member variable.
255 
256  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
257  /* %%% CONFIGURATION DEFAULT VALUES %% */
258  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
259  const auto STILLNESS_ACC_EVENT_NUMBER =
260  conf_.ul("STILLNESS_ACC_EVENT_NUMBER", 16);
261  const auto STILLNESS_GYR_EVENT_NUMBER =
262  conf_.ul("STILLNESS_GYR_EVENT_NUMBER", 32);
263  const auto STILLNESS_MAG_EVENT_NUMBER =
264  conf_.ul("STILLNESS_MAG_EVENT_NUMBER", 16);
265  const auto STILLNESS_ACC_VARIANCE_THRESHOLD =
266  conf_.d("STILLNESS_ACC_VARIANCE_THRESHOLD", 0.01);
267  const auto STILLNESS_GYR_SQUARES_THRESHOLD =
268  conf_.d("STILLNESS_GYR_SQUARES_THRESHOLD", 0.1);
269  const auto STILLNESS_MAG_VARIANCE_THRESHOLD =
270  conf_.d("STILLNESS_MAG_VARIANCE_THRESHOLD", 0.15);
271  const auto STILLNESS_GRAVITY_EPSILON =
272  conf_.d("STILLNESS_GRAVITY_EPSILON", 1.5);
273  const auto STILLNESS_MAG_FIELD_EPSILON =
274  conf_.d("STILLNESS_MAG_FIELD_EPSILON", 30.0);
275  const auto STILLNESS_SPEED_QUOCIENT =
276  conf_.d("STILLNESS_SPEED_QUOCIENT", 0.0);
277  const auto STILLNESS_UPDATE_GRAVITY =
278  conf_.ul("STILLNESS_UPDATE_GRAVITY", 1);
279  const auto ACC_DROP_INIT =
280  conf_.ul("ACC_DROP_INIT", 32);
281  const auto GYR_DROP_INIT =
282  conf_.ul("GYR_DROP_INIT", 32);
283  const auto MAG_DROP_INIT =
284  conf_.ul("MAG_DROP_INIT", 32);
285  const auto ACC_EPSILON_ZERO =
286  conf_.d("ACC_EPSILON_ZERO", 0.07);
287  const auto VEL_EPSILON_ZERO =
288  conf_.d("VEL_EPSILON_ZERO", 0.3);
289  const auto GYR_EPSILON_ZERO =
290  conf_.d("GYR_EPSILON_ZERO", 0.02);
291 
292  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
293  /* %%%%%%% STILLNESS DETECTION %%%%%%% */
294  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
295 
296  // Calibration routine
297  FunctionFilter<RawEventT> calibrate { [&](const RawEventT &) {
298  if (calibrated_)
299  return true;
300  // If all the sensors report stillness
301  if (accelerometer_stats_.size == STILLNESS_ACC_EVENT_NUMBER &&
302  accelerometer_stats_.variance < STILLNESS_ACC_VARIANCE_THRESHOLD &&
303  gyroscope_stats_.size == STILLNESS_GYR_EVENT_NUMBER &&
304  gyroscope_stats_.squares < STILLNESS_GYR_SQUARES_THRESHOLD &&
305  magnetic_field_stats_.size == STILLNESS_MAG_EVENT_NUMBER &&
306  magnetic_field_stats_.variance < STILLNESS_MAG_VARIANCE_THRESHOLD) {
307  // Store initial forces and mark as calibrated
308  gravity_.value = accelerometer_stats_.mean;
309  magnetic_field_init_.value = magnetic_field_stats_.mean;
310  calibrated_ = true;
311  }
312  return false;
313  }};
314 
315  // Fix drift routine
316  Callback<RawEventT> fix_drift { [&]() {
317  using namespace ArrayMatrix;
318  using std::abs;
319 
320  // If the accelerometer and gyroscope are still
321  if (gyroscope_stats_.squares < STILLNESS_GYR_SQUARES_THRESHOLD &&
322  accelerometer_stats_.variance < STILLNESS_ACC_VARIANCE_THRESHOLD &&
323  abs(norm(gravity_.value) - norm(accelerometer_stats_.mean)) <
324  STILLNESS_GRAVITY_EPSILON) {
325 
326  // According to the orientation matrix, we know that gravity should
327  // point towards direction A, but it points towards direction B.
328  // We fix the orientation using a rotation matrix from A to B.
329  // drift_fix * orientation * mean == gravity
330  auto drift_fix =
331  rotation_matrix(orientation_.value * accelerometer_stats_.mean,
332  gravity_.value);
333  orientation_.value = drift_fix * orientation_.value;
334  // As accelerometer is inaccurate and gravity magnitude
335  // changes significantly during rotation, resize the gravity
336  // to the current magnitude of acceleration
337  if (STILLNESS_UPDATE_GRAVITY) {
338  gravity_.value =
339  gravity_.value *
340  (accelerometer_stats_.mean_magnitude / norm(gravity_.value));
341  }
342  velocity_.value = velocity_.value * STILLNESS_SPEED_QUOCIENT;
343 
344  // If the magnetic field is also still
345  if (magnetic_field_stats_.variance < STILLNESS_MAG_VARIANCE_THRESHOLD &&
346  abs(norm(magnetic_field_stats_.mean)-norm(magnetic_field_init_.value))
347  < STILLNESS_MAG_FIELD_EPSILON) {
348  auto gravity_to_Z =
349  rotation_matrix(gravity_.value, Vector<ValueT,3>{{0,0,1}});
350  // Function to project a vector to a plane perpendicular to the gravity
351  auto project = [&](Vector<ValueT, 3> v) {
352  v = gravity_to_Z * v;
353  v[2] = 0;
354  return v * gravity_to_Z;
355  };
356  auto projected_mean =
357  project(orientation_.value * magnetic_field_stats_.mean);
358  auto projected_init =
359  project(magnetic_field_init_.value);
360  auto drift_fix =
361  rotation_matrix(projected_mean, projected_init);
362  orientation_.value = drift_fix * orientation_.value;
363  }
364  }
365  }};
366 
367  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
368  /* %%%%%% ACCELEROMETER FILTERS %%%%%% */
369  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
370  auto accelerometer_filters = std::make_tuple(
371  Callback<RawEventT>{raw_accelerometer_callback_},
372  Drop<RawEventT>{ACC_DROP_INIT},
373  Statistics<RawEventT>{&accelerometer_stats_, STILLNESS_ACC_EVENT_NUMBER},
374  calibrate,
375  fix_drift,
376  PreMultiplyBy<RawEventT, OrientationEventT>{&orientation_},
377  RemoveValueOf<RawEventT>{&gravity_},
378  EpsilonZero<RawEventT>{ACC_EPSILON_ZERO},
379  Callback<RawEventT>{filtered_accelerometer_callback_},
380  ChangeType<RawEventT>{EventType::VELOCITY},
381  DeltaIntegrate<RawEventT>{},
382  AddValueTo<RawEventT>{&velocity_},
383  EpsilonZero<RawEventT>{VEL_EPSILON_ZERO},
384  Callback<RawEventT>{velocity_callback_},
385  ChangeType<RawEventT>{EventType::DISPLACEMENT},
386  DeltaIntegrate<RawEventT>{},
387  AddValueTo<RawEventT>{&displacement_},
388  Callback<RawEventT>{displacement_callback_}
389  );
390  connect_pipeline(accelerometer_filters);
391 
392  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
393  /* %%%%%%%% GYROSCOPE FILTERS %%%%%%%% */
394  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
395  auto gyroscope_filters = std::make_tuple(
396  Callback<RawEventT>{raw_gyroscope_callback_},
397  Drop<RawEventT>{GYR_DROP_INIT},
398  Statistics<RawEventT>{&gyroscope_stats_, STILLNESS_GYR_EVENT_NUMBER},
399  calibrate,
400  fix_drift,
401  EpsilonZero<RawEventT>{GYR_EPSILON_ZERO},
402  Callback<RawEventT>{filtered_gyroscope_callback_},
403  ChangeType<RawEventT>{EventType::ORIENTATION},
404  DeltaIntegrate<RawEventT>{},
405  ToRotationMatrix<RawEventT,OrientationEventT>{},
406  PostMultiplyThe<OrientationEventT>{&orientation_},
407  Callback<OrientationEventT>{orientation_callback_}
408  );
409  connect_pipeline(gyroscope_filters);
410 
411  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
412  /* %%%%%% MAGNETIC FIELD FILTERS %%%%% */
413  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
414  auto magnetic_field_filters = std::make_tuple(
415  Callback<RawEventT>{raw_magnetic_field_callback_},
416  Drop<RawEventT>{MAG_DROP_INIT},
417  Statistics<RawEventT>{&magnetic_field_stats_, STILLNESS_MAG_EVENT_NUMBER},
418  calibrate,
419  fix_drift,
420  PreMultiplyBy<RawEventT, OrientationEventT>{&orientation_},
421  RemoveValueOf<RawEventT>{&magnetic_field_init_},
422  Callback<RawEventT>{filtered_magnetic_field_callback_}
423  );
424  connect_pipeline(magnetic_field_filters);
425 
426  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
427  /* %%% LINEAR ACCELERATION FILTERS %%% */
428  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
429  // Linear acceleration and accelerometer + gyroscope
430  // are mutually exclusive.
431  auto linear_acceleration_filters = std::make_tuple(
432  ChangeType<RawEventT>{EventType::ACCELEROMETER},
433  Callback<RawEventT>{raw_linear_accelerometer_callback_},
434  Drop<RawEventT>{ACC_DROP_INIT},
435  Callback<RawEventT>{filtered_linear_accelerometer_callback_},
436  ChangeType<RawEventT>{EventType::VELOCITY},
437  DeltaIntegrate<RawEventT>{},
438  AddValueTo<RawEventT>{&velocity_},
439  Callback<RawEventT>{linear_velocity_callback_},
440  ChangeType<RawEventT>{EventType::DISPLACEMENT},
441  DeltaIntegrate<RawEventT>{},
442  AddValueTo<RawEventT>{&displacement_},
443  Callback<RawEventT>{linear_displacement_callback_}
444  );
445  connect_pipeline(linear_acceleration_filters);
446 
447  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
448  /* %%%%%%%%%%%% MAIN LOOP %%%%%%%%%%%% */
449  /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
450  for (auto event = events_.pop_get_front();
451  event.type != EventType::STOP;
452  event = events_.pop_get_front()) {
453  std::lock_guard<std::mutex> lock(mutex_);
454  switch (event.type) {
455  case EventType::ACCELEROMETER:
456  std::get<0>(accelerometer_filters).receive(event);
457  break;
458  case EventType::GYROSCOPE:
459  std::get<0>(gyroscope_filters).receive(event);
460  break;
461  case EventType::MAGNETIC_FIELD:
462  std::get<0>(magnetic_field_filters).receive(event);
463  break;
464  case EventType::LINEAR_ACCELERATION:
465  std::get<0>(linear_acceleration_filters).receive(event);
466  break;
467  default:;
468  }
469  }
470 }
